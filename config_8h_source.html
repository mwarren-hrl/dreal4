151  // --------------------------------------------------------------------------
152  // NLopt options (stopping criteria)
153  // --------------------------------------------------------------------------
154  // These options are used when we use NLopt in refining
155  // counterexamples via local-optimization. The following
156  // descriptions are from
157  // https://nlopt.readthedocs.io/en/latest/NLopt_Reference/#stopping-criteria
158  //
159  // Set relative tolerance on function value: stop when an
160  // optimization step (or an estimate of the optimum) changes the
161  // objective function value by less than tol multiplied by the
162  // absolute value of the function value. (If there is any chance
163  // that your optimum function value is close to zero, you might want
164  // to set an absolute tolerance with nlopt_set_ftol_abs as well.)
165  // Criterion is disabled if tol is non-positive.
166  OptionValue<double> nlopt_ftol_rel_{kDefaultNloptFtolRel};
167 
168  // Set absolute tolerance on function value: stop when an
169  // optimization step (or an estimate of the optimum) changes the
170  // function value by less than tol. Criterion is disabled if tol is
171  // non-positive.
172  OptionValue<double> nlopt_ftol_abs_{kDefaultNloptFtolAbs};
173 
174  // Stop when the number of function evaluations exceeds
175  // maxeval. (This is not a strict maximum: the number of function
176  // evaluations may exceed maxeval slightly, depending upon the
177  // algorithm.) Criterion is disabled if maxeval is non-positive.
178  OptionValue<int> nlopt_maxeval_{kDefaultNloptMaxEval};
179 
180  // Stop when the optimization time (in seconds) exceeds
181  // maxtime. (This is not a strict maximum: the time may exceed
182  // maxtime slightly, depending upon the algorithm and on how slow
183  // your function evaluation is.) Criterion is disabled if maxtime is
184  // non-positive.
185  OptionValue<double> nlopt_maxtime_{kDefaultNloptMaxTime};
186 
187  // Default initial phase (for PICOSAT):
188  // 0 = false
189  // 1 = true
190  // 2 = Jeroslow-Wang (default)
191  // 3 = random initial phase
192  OptionValue<SatDefaultPhase> sat_default_phase_{
193  SatDefaultPhase::JeroslowWang};
194 
195  // Seed for Random Number Generator.
