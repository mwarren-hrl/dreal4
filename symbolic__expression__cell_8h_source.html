20 /** Represents an abstract class which is the base of concrete

64  */
65  virtual Expression Substitute(const ExpressionSubstitution& expr_subst,
66  const FormulaSubstitution& formula_subst) = 0;
67 
68  /** Differentiates this symbolic expression with respect to the variable @p
69  * var.
70  * @throws std::runtime_error if it is not differentiable.
71  */
72  virtual Expression Differentiate(const Variable& x) const = 0;
73 
74  /** Outputs string representation of expression into output stream @p os. */
75  virtual std::ostream& Display(std::ostream& os) const = 0;
76 
77  /** Returns the reference count of this cell. */
78  unsigned use_count() const {

82  /** Copy-constructs an ExpressionCell from an lvalue. (DELETED) */
83  ExpressionCell(const ExpressionCell& e) = delete;
84

86  ExpressionCell(ExpressionCell&& e) = delete;

92  ExpressionCell& operator=(const ExpressionCell& e) = delete;
93 
94  protected:

97  /** Constructs ExpressionCell of kind @p k with @p hash, @p is_poly, and @p
98  * include_ite. */
99  ExpressionCell(ExpressionKind k, size_t hash, bool is_poly, bool include_ite,
100  Variables variables);

102  virtual ~ExpressionCell() = default;
103  /** Returns an expression pointing to this ExpressionCell. */
104  Expression GetExpression();
105 
106  private:
107  const ExpressionKind kind_{};
108  const size_t hash_{};
109  const bool is_polynomial_{false};
110  const bool include_ite_{false};
111  const Variables variables_;
112 
113  // Reference counter.
114  mutable std::atomic<unsigned> rc_{0};
115  void increase_rc() const {
116  atomic_fetch_add_explicit(&rc_, 1U, std::memory_order_relaxed);
117  }
118  void decrease_rc() const {
119  if (atomic_fetch_sub_explicit(&rc_, 1U, std::memory_order_acq_rel) == 1U) {
