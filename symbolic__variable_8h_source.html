34  /** Default constructor. Constructs a dummy variable of CONTINUOUS type. This
35  * is needed to have Eigen::Matrix<Variable>. The objects created by the
36  * default constructor share the same ID, zero. As a result, they all are
37  * identified as a single variable by equality operator (==). They all have
38  * the same hash value as well.
39  *
40  * It is allowed to construct a dummy variable but it should not be used to
41  * construct a symbolic expression.
42  */

46  name_{std::make_shared<std::string>()} {}

59  /** Checks if this is a dummy variable (ID = 0) which is created by

61  bool is_dummy() const { return get_id() == 0; }
62  Id get_id() const;
63  Type get_type() const;
64  size_t get_hash() const { return std::hash<Id>{}(id_); }
65  std::string get_name() const;

67 
68  /// Checks the equality of two variables based on their ID values.
69  bool equal_to(const Variable& v) const { return get_id() == v.get_id(); }
70 
71  /// Compares two variables based on their ID values.
72  bool less(const Variable& v) const { return get_id() < v.get_id(); }
